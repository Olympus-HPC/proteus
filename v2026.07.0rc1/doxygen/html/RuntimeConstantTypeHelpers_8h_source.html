21inline RuntimeConstantType convertTypeToRuntimeConstantType(Type *Ty) {
22 if (Ty->isIntegerTy(1))
23 return RuntimeConstantType::BOOL;
24 if (Ty->isIntegerTy(8))
25 return RuntimeConstantType::INT8;
26 if (Ty->isIntegerTy(32))
27 return RuntimeConstantType::INT32;
28 if (Ty->isIntegerTy(64))
29 return RuntimeConstantType::INT64;
30 if (Ty->isFloatTy())
31 return RuntimeConstantType::FLOAT;
32 if (Ty->isDoubleTy())
33 return RuntimeConstantType::DOUBLE;
34 if (Ty->isFP128Ty() || Ty->isPPC_FP128Ty() || Ty->isX86_FP80Ty())
35 return RuntimeConstantType::LONG_DOUBLE;
36 if (Ty->isPointerTy())
