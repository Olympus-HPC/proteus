192codegenParallel(Module &M, StringRef DeviceArch,
193 const OptimizationPipelineConfig &OptConfig =
194 OptimizationPipelineConfig(std::nullopt, '3', 3)) {
195 TIMESCOPE("proteus::codegenParallel");
196 // Use regular LTO with parallelism enabled to parallelize codegen.
197 std::atomic<bool> LTOError = false;
198
199 auto DiagnosticHandler = [&](const DiagnosticInfo &DI) {
200 std::string ErrStorage;
201 raw_string_ostream OS(ErrStorage);
202 DiagnosticPrinterRawOStream DP(OS);
203 DI.print(DP);
204
205 switch (DI.getSeverity()) {
206 case DS_Error:
207 WithColor::error(errs(), "[proteus codegen]") << ErrStorage << "\n";
208 LTOError = true;
209 break;
210 case DS_Warning:
211 WithColor::warning(errs(), "[proteus codegen]") << ErrStorage << "\n";
212 break;
213 case DS_Note:
214 WithColor::note(errs(), "[proteus codegen]") << ErrStorage << "\n";
215 break;
216 case DS_Remark:
217 WithColor::remark(errs()) << ErrStorage << "\n";
218 break;
219 }
220 };
221
222 // Create TargetMachine and extract options/features.
223 auto ExpectedTM = proteus::detail::createTargetMachine(
224 M, DeviceArch, OptConfig.CodegenOptLevel);
225 if (!ExpectedTM)
226 reportFatalError(toString(ExpectedTM.takeError()));
227 std::unique_ptr<TargetMachine> TM = std::move(*ExpectedTM);
228
229 lto::Config Conf;
230 Conf.CPU = DeviceArch;
231
232 // Propagate attributes from TargetMachine to LTO Config.
233 std::string FeatureStr = TM->getMCSubtargetInfo()->getFeatureString().str();
234 if (!FeatureStr.empty()) {
235 SmallVector<StringRef> Features;
236 StringRef(FeatureStr).split(Features, ',');
237 for (auto &F : Features)
238 Conf.MAttrs.push_back(F.str());
239 } else {
240 Conf.MAttrs = {};
241 }
242
243 // FIX: Propagate TargetOptions (e.g. UnsafeFPMath, etc.)
244 Conf.Options = TM->Options;
245
246 Conf.DisableVerify = true;
247 Conf.TimeTraceEnabled = false;
248 Conf.DebugPassManager = false;
249 Conf.VerifyEach = false;
250 Conf.DiagHandler = DiagnosticHandler;
251 Conf.OptLevel = OptConfig.OptLevel;
252 const auto Plugins = getJITPassPluginConfigs();
253 // Parallel codegen lets LTO own optimization, so custom textual pipelines
254 // must be forwarded to the LTO configuration instead of run beforehand.
255 if (OptConfig.PassPipeline || !Plugins.empty())
256 Conf.OptPipeline = proteus::detail::composeOptimizationPassPipeline(
257 OptConfig.PassPipeline, OptConfig.OptLevel, Plugins);
258 for (const auto &Plugin : Plugins)
259 Conf.PassPlugins.push_back(Plugin.Path);
260 Conf.CGOptLevel = static_cast<CodeGenOptLevel>(OptConfig.CodegenOptLevel);
261
262 unsigned ParallelCodeGenParallelismLevel =
263 std::max(1u, std::thread::hardware_concurrency());
264 lto::LTO L(std::move(Conf), {}, ParallelCodeGenParallelismLevel);
265
266 // Ensure module has the correct DataLayout prior to emitting bitcode.
267 M.setDataLayout(TM->createDataLayout());
268
269 SmallString<0> BitcodeBuf;
270 raw_svector_ostream BitcodeOS(BitcodeBuf);
271 WriteBitcodeToFile(M, BitcodeOS);
272
273 // TODO: Module identifier can be empty because you always have on module to
274 // link. However, in the general case, with multiple modules, each one must
275 // have a unique identifier for LTO to work correctly.
276 auto IF = cantFail(lto::InputFile::create(
277 MemoryBufferRef{BitcodeBuf, M.getModuleIdentifier()}));
278
279 std::set<std::string> PrevailingSymbols;
280 auto BuildResolutions = [&]() {
281 // Save the input file and the buffer associated with its memory.
282 const auto Symbols = IF->symbols();
283 SmallVector<lto::SymbolResolution, 16> Resolutions(Symbols.size());
284 size_t SymbolIdx = 0;
285 for (auto &Sym : Symbols) {
286 lto::SymbolResolution &Res = Resolutions[SymbolIdx];
287 SymbolIdx++;
288
289 // All defined symbols are prevailing.
290 Res.Prevailing = !Sym.isUndefined() &&
291 PrevailingSymbols.insert(Sym.getName().str()).second;
292
293 Res.VisibleToRegularObj =
294 Res.Prevailing &&
295 Sym.getVisibility() != GlobalValue::HiddenVisibility &&
296 !Sym.canBeOmittedFromSymbolTable();
297
298 Res.ExportDynamic =
299 Sym.getVisibility() != GlobalValue::HiddenVisibility &&
300 (!Sym.canBeOmittedFromSymbolTable());
301
302 Res.FinalDefinitionInLinkageUnit =
303 Sym.getVisibility() != GlobalValue::DefaultVisibility &&
304 (!Sym.isUndefined() && !Sym.isCommon());
305
306 // Device linking does not support linker redefined symbols (e.g.
307 // --wrap).
308 Res.LinkerRedefined = false;
309
310 if (Config::get().ProteusDebugOutput) {
311 auto PrintSymbol = [](const lto::InputFile::Symbol &Sym,
312 lto::SymbolResolution &Res) {
313 auto &OutStream = Logger::logs("proteus");
314 OutStream << "Vis: ";
315 switch (Sym.getVisibility()) {
316 case GlobalValue::HiddenVisibility:
317 OutStream << 'H';
318 break;
319 case GlobalValue::ProtectedVisibility:
320 OutStream << 'P';
321 break;
322 case GlobalValue::DefaultVisibility:
323 OutStream << 'D';
324 break;
325 }
326
327 OutStream << " Sym: ";
328 auto PrintBool = [&](char C, bool B) { OutStream << (B ? C : '-'); };
329 PrintBool('U', Sym.isUndefined());
330 PrintBool('C', Sym.isCommon());
331 PrintBool('W', Sym.isWeak());
332 PrintBool('I', Sym.isIndirect());
333 PrintBool('O', Sym.canBeOmittedFromSymbolTable());
334 PrintBool('T', Sym.isTLS());
335 PrintBool('X', Sym.isExecutable());
336 OutStream << ' ' << Sym.getName();
337 OutStream << "| P " << Res.Prevailing;
338 OutStream << " V " << Res.VisibleToRegularObj;
339 OutStream << " E " << Res.ExportDynamic;
340 OutStream << " F " << Res.FinalDefinitionInLinkageUnit;
341 OutStream << "\n";
342 };
343
344 PrintSymbol(Sym, Res);
345 }
346 }
347
348 // Add the bitcode file with its resolved symbols to the LTO job.
349 cantFail(L.add(std::move(IF), Resolutions));
350 };
351
352 BuildResolutions();
353
354 // Run the LTO job to compile the bitcode.
355 size_t MaxTasks = L.getMaxTasks();
356 SmallVector<std::unique_ptr<sys::fs::TempFile>> ObjectFiles{MaxTasks};
357
358 auto AddStream =
359 [&](size_t Task,
360 const Twine & /*ModuleName*/) -> std::unique_ptr<CachedFileStream> {
361 std::string TaskStr = Task ? "." + std::to_string(Task) : "";
362 auto ExpectedF = createTempFile("lto-shard" + TaskStr, "o");
363 if (auto E = ExpectedF.takeError())
364 reportFatalError("Error creating tmp file " + toString(std::move(E)));
365 ObjectFiles[Task] =
366 std::make_unique<sys::fs::TempFile>(std::move(*ExpectedF));
367 auto Ret = std::make_unique<CachedFileStream>(
368 std::make_unique<llvm::raw_fd_ostream>(ObjectFiles[Task]->FD, false));
369 if (!Ret)
370 reportFatalError("Error creating CachedFileStream");
371 return Ret;
372 };
373
374 if (Error E = L.run(AddStream))
375 reportFatalError("Error: " + toString(std::move(E)));
376
377 if (LTOError)
378 reportFatalError(toString(
379 createStringError(inconvertibleErrorCode(),
380 "Errors encountered inside the LTO pipeline.")));
381
382 return ObjectFiles;
383}

386inline std::unique_ptr<MemoryBuffer> codegenRTC(Module &M,
387 StringRef DeviceArch) {
388 TIMESCOPE("proteus::codegenRTC");
389 char *BinOut;
390 size_t BinSize;
391
392 SmallString<4096> ModuleBuf;
393 raw_svector_ostream ModuleBufOS(ModuleBuf);
394 WriteBitcodeToFile(M, ModuleBufOS);
395
396 hiprtcLinkState HipLinkStatePtr;
397
398 // NOTE: This code is an example of passing custom, AMD-specific
399 // options to the compiler/linker.
400 // NOTE: Unrolling can have a dramatic (time-consuming) effect on JIT
401 // compilation time and on the resulting optimization, better or worse
402 // depending on code specifics.
403 std::string MArchOpt = ("-march=" + DeviceArch).str();
404
405 // NOTE: We used to pass these options as well. "-mllvm",
406 // "-unroll-threshold=1000",
407 // We removed them cause we saw on bezier they cause slowdowns
408 const char *OptArgs[] = {MArchOpt.c_str()};
409 std::vector<hiprtcJIT_option> JITOptions = {
410 HIPRTC_JIT_IR_TO_ISA_OPT_EXT, HIPRTC_JIT_IR_TO_ISA_OPT_COUNT_EXT};
411 size_t OptArgsSize = 1;
412 const void *JITOptionsValues[] = {(void *)OptArgs, (void *)(OptArgsSize)};
413 proteusHiprtcErrCheck(proteus::hipdyn::rtcLinkCreate(
414 JITOptions.size(), JITOptions.data(), (void **)JITOptionsValues,
415 &HipLinkStatePtr));
416 // NOTE: the following version of te code does not set options.
417 // proteusHiprtcErrCheck(hiprtcLinkCreate(0, nullptr, nullptr,
418 // &hip_link_state_ptr));
419
420 proteusHiprtcErrCheck(proteus::hipdyn::rtcLinkAddData(
421 HipLinkStatePtr, HIPRTC_JIT_INPUT_LLVM_BITCODE, (void *)ModuleBuf.data(),
422 ModuleBuf.size(), "", 0, nullptr, nullptr));
423 proteusHiprtcErrCheck(proteus::hipdyn::rtcLinkComplete(
424 HipLinkStatePtr, (void **)&BinOut, &BinSize));
425
426 return MemoryBuffer::getMemBuffer(StringRef{BinOut, BinSize});
427}

454inline std::unique_ptr<MemoryBuffer>
455codegenObject(Module &M, StringRef DeviceArch,
456 [[maybe_unused]] SmallPtrSetImpl<void *> &GlobalLinkedBinaries,
457 CodegenOption CGOption = CodegenOption::RTC,
458 const OptimizationPipelineConfig &OptConfig =
459 OptimizationPipelineConfig(std::nullopt, '3', 3)) {
460 TIMESCOPE("proteus::codegenObjectHIP");
461 assert(GlobalLinkedBinaries.empty() &&
462 "Expected empty linked binaries for HIP");
463 Timer T(Config::get().ProteusEnableTimers);
464 SmallVector<std::unique_ptr<sys::fs::TempFile>> ObjectFiles;
465 switch (CGOption) {
466 case CodegenOption::RTC: {
467 auto Ret = detail::codegenRTC(M, DeviceArch);
468 PROTEUS_TIMER_OUTPUT(Logger::outs("proteus")
469 << "Codegen RTC " << T.elapsed() << " ms\n");
470 return Ret;
471 }
472#if LLVM_VERSION_MAJOR >= 18
473 case CodegenOption::Serial:
474 ObjectFiles = detail::codegenSerial(M, DeviceArch);
475 break;
476 case CodegenOption::Parallel:
477 ObjectFiles = detail::codegenParallel(M, DeviceArch, OptConfig);
478 break;
479#endif
480 default:
481 reportFatalError("Unknown Codegen Option");
482 }
483
484 if (ObjectFiles.empty())
485 reportFatalError("Expected non-empty vector of object files");
486
487#if LLVM_VERSION_MAJOR >= 18
488 auto ExpectedF = detail::createTempFile("proteus-jit", "o");
489 if (auto E = ExpectedF.takeError())
490 reportFatalError("Error creating shared object file " +
491 toString(std::move(E)));
492
493 auto SharedObject = std::move(*ExpectedF);
494
495 std::vector<const char *> Args{"ld.lld", "--no-undefined", "-shared", "-o",
496 SharedObject.TmpName.c_str()};
497 for (auto &File : ObjectFiles) {
498 if (!File)
499 continue;
500 Args.push_back(File->TmpName.c_str());
501 }
502
503 if (Config::get().ProteusDebugOutput) {
504 for (auto &Arg : Args) {
505 Logger::logs("proteus") << Arg << " ";
506 }
507 Logger::logs("proteus") << "\n";
508 }
509
510 PROTEUS_TIMER_OUTPUT(Logger::outs("proteus")
511 << "Codegen object " << toString(CGOption) << "["
512 << ObjectFiles.size() << "] " << T.elapsed() << " ms\n");
513
514 T.reset();
515 // The LLD linker interface is not thread-safe, so we use a mutex.
516 static std::mutex Mutex;
517 {
518 std::lock_guard LockGuard{Mutex};
519 lld::Result S = lld::lldMain(Args, llvm::outs(), llvm::errs(),
520 {{lld::Gnu, &lld::elf::link}});
521 if (S.retCode)
522 reportFatalError("Error: lld failed");
523 }
524
525 ErrorOr<std::unique_ptr<MemoryBuffer>> Buffer =
526 MemoryBuffer::getFileAsStream(SharedObject.TmpName);
527 if (!Buffer)
528 reportFatalError("Error reading file: " + Buffer.getError().message());
529
530 // Remove temporary files.
531 for (auto &File : ObjectFiles) {
532 if (!File)
533 continue;
534 if (auto E = File->discard())
535 reportFatalError("Error removing object tmp file " +
536 toString(std::move(E)));
537 }
538 if (auto E = SharedObject.discard())
539 reportFatalError("Error removing shared object tmp file " +
540 toString(std::move(E)));
541
542 PROTEUS_TIMER_OUTPUT(Logger::outs("proteus")
543 << "Codegen linking " << T.elapsed() << " ms\n");
544
545 return std::move(*Buffer);
546#else
547 reportFatalError("Expected LLVM18 for non-RTC codegen");
548#endif
549}
